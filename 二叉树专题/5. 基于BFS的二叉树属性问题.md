# 基于BFS的二叉树属性问题

## 二叉树的最大最小深度

### 思路
* 通过BFS遍历每一层，知道某一层的子节点个数为0；

### 代码
```class Solution {
public:
    queue<TreeNode*> q;
    int deep = 0;

    int minDepth(TreeNode* root) {
        if(root) q.push(root);
        while(!q.empty()){
            int s = q.size();
            deep++;
            for(int i=0; i<s; i++){
                TreeNode* node = q.front();
                q.pop();
                if(!node->left && !node->right) return deep;
                if(node->left) q.push(node->left);
                if(node->right) q.push(node->right);
            }
        }
        return deep;
    }
};
```

```
class Solution {
    queue<TreeNode*> q;
    int max_deep = 0;

public:
    int maxDepth(TreeNode* root) {
        if(root) q.push(root);
        while(!q.empty()){
            int s = q.size();
            while(s--){
                TreeNode* n = q.front();
                q.pop();
                if(n->left) q.push(n->left);
                if(n->right) q.push(n->right);
            }
            max_deep++;
        }
        return max_deep;
    }
};
```
